OBSERVER AUTOMATON AssumptionAutomaton

INITIAL STATE A0;

STATE __TRUE :
    TRUE -> GOTO __TRUE;

STATE __qTEMP :
    TRUE -> ASSUME {false} GOTO __qTEMP;

STATE __qERROR :
    TRUE -> ASSUME {true} GOTO __qERROR;

STATE USEFIRST A0 :
    MATCH "int main()" -> GOTO A2_8_1;
    TRUE -> GOTO A0;

STATE USEFIRST A10 :
    MATCH "__VERIFIER_assert((i+j) == l);" -> GOTO A11;
    TRUE -> GOTO A10;

STATE USEFIRST A11 :
    MATCH "[!(cond)]" -> GOTO A13;
    MATCH "[!(!(cond))]" -> GOTO __qTEMP;
    TRUE -> GOTO A11;

STATE USEFIRST A13 :
    MATCH "reach_error();" -> GOTO __qERROR;
    TRUE -> GOTO A13;

STATE USEFIRST A2_8_1 :
    MATCH "[!(n <= SIZE)]" -> GOTO __qTEMP;
    MATCH "[n <= SIZE]" -> GOTO A3;
    TRUE -> GOTO A2_8_1;

STATE USEFIRST A3 :
    MATCH "l=0;" -> GOTO A6;
    TRUE -> GOTO A3;

STATE USEFIRST A6 :
    MATCH "[!(l < n)]" -> GOTO A10;
    MATCH "[l < n]" -> GOTO __qTEMP;
    TRUE -> GOTO A6;

END AUTOMATON
